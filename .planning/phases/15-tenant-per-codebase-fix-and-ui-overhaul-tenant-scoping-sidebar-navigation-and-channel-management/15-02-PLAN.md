**1. Add message grouping logic in MessageFeed.tsx:**

Before rendering, compute which messages are "grouped" (continuation of same sender within 5 minutes):

```tsx
// In MessageFeed component, after filtering topLevelMessages:
const groupedMessages = useMemo(() => {
  return topLevelMessages.map((msg, i) => {
    if (i === 0) return { message: msg, isGrouped: false, showDateSeparator: true };
    const prev = topLevelMessages[i - 1]!;
    const sameSender = msg.senderId === prev.senderId;
    const timeDiff = new Date(msg.createdAt).getTime() - new Date(prev.createdAt).getTime();
    const within5Min = timeDiff < 5 * 60 * 1000;
    const isGrouped = sameSender && within5Min && msg.senderType !== 'system';

    // Date separator: show if the day changes
    const msgDate = new Date(msg.createdAt).toDateString();
    const prevDate = new Date(prev.createdAt).toDateString();
    const showDateSeparator = msgDate !== prevDate;

    return { message: msg, isGrouped: isGrouped && !showDateSeparator, showDateSeparator };
  });
}, [topLevelMessages]);
```

**2. Add DateSeparator rendering in MessageFeed.tsx:**

```tsx
function formatDateSeparator(dateStr: string): string {
  const date = new Date(dateStr);
  const today = new Date();
  const yesterday = new Date(today);
  yesterday.setDate(yesterday.getDate() - 1);

  if (date.toDateString() === today.toDateString()) return 'Today';
  if (date.toDateString() === yesterday.toDateString()) return 'Yesterday';
  return date.toLocaleDateString('en-US', { weekday: 'long', month: 'long', day: 'numeric', year: 'numeric' });
}
```
